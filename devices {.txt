sql query to create table devices contains name,type and table parameters contains all parameters:

create table devices(id int auto_increment, 
                      type varchar(50) not null,
                      name varchar(100) not null,
                      primary key(id));
create table parameters(id int auto_increment,parameter_id int,
                        isOn bool, 
                        isOpen bool,
                        temp int,
                        frequency float, 
                        volume int,
                        timer int,
                        mode varchar(50),
                        primary key (id),
                        foreign key(parameter_id) references devices(id));
